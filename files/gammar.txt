MAIN → STMT | FLIST | ε

FLIST  → FDEF FLIST'
FLIST' → FLIST | e

FDEF → def id(PARLIST){STMTLIST}

PARLIST  → int id PARLIST' | e
PARLIST' → , PARLIST | e

STMT → int id;
 | ATRIBST;
 | PRINTST;
 | RETURNST;
 | IFSTMT
 | {STMTLIST}
 | ;

ATRIBST' → id TESTE | FACTOR' TERM' NUMEXPR' EXPR' 
TESTE    → (PARLISTCAL) | TERM' NUMEXPR' EXPR'  
FACTOR'  → num | (NUMEXPR)

FCALL → id(PARLISTCALL) # Inutilizou, poderia ser removida.

PARLISTCALL  → id PARLISTCALL' | e
PARLISTCALL' → , PARLISTCALL | e

PRINTST → print EXPR

RETURNST → return

IFSTMT  → if(EXPR) STMT IFSTMT'
IFSTMT' → else STMT | e 

STMLIST  → STMT STMLIST'
STMLIST' → STMLIST | e

EXPR  → NUMEXPR EXPR'

 | > NUMEXPR
 | == NUMEXPR
 | e

NUMEXPR' → + TERM NUMEXPR' 
 | - TERM NUMEXPR'
 | e

TERM  → FACTOR TERM'
TERM' → *FACTOR TERM' | e
